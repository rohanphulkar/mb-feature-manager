const fetchConfig = async () => {
  const appId = "malibandhu-app";
  const timestamp = Date.now().toString();
  
  // Generate SHA256 Signature (HMAC)
  const signature = generateSignature(appId, timestamp, SECRET);

  const response = await fetch('/api/config', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ appId, timestamp, signature })
  });

  return await response.json();
};
